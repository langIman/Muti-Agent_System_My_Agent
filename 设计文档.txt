  核心模块设计

  1. 全局状态 (orchestrator/state.py)

  from typing import TypedDict, Annotated
  from langgraph.graph.message import add_messages

  class AgentState(TypedDict):
      messages: Annotated[list, add_messages]
      context: dict          # 环境感知结果
      plan: list[dict]       # 任务计划 [{step, action, status}]
      current_step: int
      memory_query: str      # 记忆检索关键词
      memory_result: str     # 记忆检索结果
      tool_results: list     # 工具执行结果
      feedback: dict         # 学习反馈

  2. 主控调度器 (orchestrator/core.py)

  用 LangGraph 的 StateGraph 构建 Agent 间的流转：

  from langgraph.graph import StateGraph, END
  from orchestrator.state import AgentState

  def build_graph():
      graph = StateGraph(AgentState)

      # 注册节点 — 每个节点对应一个 Agent
      graph.add_node("perceive", perception_agent)
      graph.add_node("recall", memory_agent)
      graph.add_node("plan", planner_agent)
      graph.add_node("execute", executor_agent)
      graph.add_node("learn", learner_agent)

      # 定义流转
      graph.set_entry_point("perceive")
      graph.add_edge("perceive", "recall")
      graph.add_edge("recall", "plan")
      graph.add_edge("plan", "execute")
      graph.add_conditional_edges("execute", should_continue, {
          "next_step": "execute",
          "replan":    "plan",
          "learn":     "learn",
      })
      graph.add_edge("learn", END)

      return graph.compile()

  流转逻辑：

  用户输入 → 环境感知 → 记忆检索 → 任务规划 → 逐步执行 ─┐
                                      ↑                    │
                                      └── 需要重规划 ───────┘
                                                           │
                                                执行完成 → 学习 → 结束

  3. Agent 基类 (agents/base.py)

  from langchain_openai import ChatOpenAI
  from langchain_core.prompts import ChatPromptTemplate

  class BaseAgent:
      def __init__(self, name: str, system_prompt: str, model: str = "gpt-4o"):
          self.name = name
          self.llm = ChatOpenAI(model=model)
          self.prompt = ChatPromptTemplate.from_messages([
              ("system", system_prompt),
              ("placeholder", "{messages}"),
          ])
          self.chain = self.prompt | self.llm

      def __call__(self, state: dict) -> dict:
          response = self.chain.invoke(state)
          return {"messages": [response]}

  4. 五大核心 Agent

  (a) 环境感知 Agent — agents/perception.py

  职责：分析用户输入，收集上下文（时间、文件内容、系统状态等）

  from agents.base import BaseAgent
  from langchain_core.tools import tool

  @tool
  def read_file(path: str) -> str:
      """读取本地文件内容"""
      with open(path) as f:
          return f.read()

  @tool
  def get_system_info() -> dict:
      """获取系统环境信息"""
      import platform
      return {"os": platform.system(), "python": platform.python_version()}

  class PerceptionAgent(BaseAgent):
      def __init__(self):
          super().__init__(
              name="Perceiver",
              system_prompt=(
                  "你是环境感知专家。分析用户意图，调用工具收集必要的上下文信息，"
                  "输出结构化的 context dict，包含: intent, entities, env_info。"
              ),
          )
          self.llm = self.llm.bind_tools([read_file, get_system_info])

      def __call__(self, state):
          result = self.chain.invoke(state)
          # 解析 tool_calls 并执行，将结果写入 state["context"]
          context = self._run_tools(result)
          return {"messages": [result], "context": context}

  (c) 工具调用 Agent — agents/executor.py

  职责：按计划逐步执行，调用具体工具

  from langchain_core.tools import tool
  from langchain.agents import create_tool_calling_agent, AgentExecutor

  class ExecutorAgent(BaseAgent):
      def __init__(self, tools: list):
          super().__init__(
              name="Executor",
              system_prompt="你是执行专家。严格按照当前 step 的指令调用工具并返回结果。",
          )
          self.tools = tools
          self.llm = self.llm.bind_tools(tools)

      def __call__(self, state):
          step = state["plan"][state["current_step"]]
          # 将当前步骤注入 messages，让 LLM 决定调用哪个 tool
          result = self.chain.invoke({
              **state,
              "messages": state["messages"] + [("user", f"执行: {step}")],
          })
          tool_output = self._run_tools(result)
          return {
              "messages": [result],
              "current_step": state["current_step"] + 1,
              "tool_results": state.get("tool_results", []) + [tool_output],
          }

  条件路由函数：

  def should_continue(state) -> str:
      if state["current_step"] < len(state["plan"]):
          last_result = state["tool_results"][-1]
          if last_result.get("error"):
              return "replan"   # 出错 → 重新规划
          return "next_step"    # 继续下一步
      return "learn"            # 全部完成 → 进入学习

  三层记忆架构：

  ┌──────────┬──────────────────────────────────────┬──────────────────┐
  │   层级   │                 实现                 │       用途       │
  ├──────────┼──────────────────────────────────────┼──────────────────┤

  ├──────────┼──────────────────────────────────────┼──────────────────┤
  │ 长期记忆 │ Chroma 向量数据库                    │ 知识/经验持久化  │
  ├──────────┼──────────────────────────────────────┼──────────────────┤
  │ 情景记忆 │ SQLite                               │ 任务执行历史记录 │
  └──────────┴──────────────────────────────────────┴──────────────────┘

  from langchain_chroma import Chroma
  from langchain_openai import OpenAIEmbeddings

  class MemoryAgent(BaseAgent):
      def __init__(self):
          super().__init__(name="Memory", system_prompt="你是记忆管理专家。")
          self.vectorstore = Chroma(
              collection_name="long_term",
              embedding_function=OpenAIEmbeddings(),
              persist_directory="./data/chroma",
          )
          self.retriever = self.vectorstore.as_retriever(search_kwargs={"k": 5})

      def __call__(self, state):
          # 检索相关记忆
          query = state.get("context", {}).get("intent", "")
          docs = self.retriever.invoke(query)
          memory_result = "\n".join(d.page_content for d in docs)
          return {"memory_result": memory_result}

      def store(self, text: str, metadata: dict):
          """存储新记忆"""
          self.vectorstore.add_texts([text], metadatas=[metadata])

  (e) 持续学习 Agent — agents/learner.py

  职责：从执行结果中提取经验，优化未来行为

  class LearnerAgent(BaseAgent):
      def __init__(self, memory_agent: MemoryAgent):
          super().__init__(
              name="Learner",
              system_prompt=(
                  "你是学习专家。分析任务执行过程，提取：\n"
                  "1. 成功策略（可复用的模式）\n"
                  "2. 失败教训（应避免的做法）\n"
                  "3. Prompt 改进建议\n"
                  "输出 JSON: {\"lessons\": [...], \"strategy\": \"...\", \"prompt_patch\": \"...\"}"
              ),
          )
          self.memory = memory_agent

      def __call__(self, state):
          result = self.chain.invoke(state)
          lessons = parse_json(result.content)

          # 将经验存入长期记忆
          for lesson in lessons.get("lessons", []):
              self.memory.store(lesson, metadata={"type": "lesson"})

          return {"messages": [result], "feedback": lessons}

  关键设计决策

  1. LangGraph 而非 AgentExecutor — LangGraph 的 StateGraph 提供显式的状态流转控制，比 AgentExecutor
  的隐式循环更可控、可调试
  2. Agent 间通过 State 通信 — 不用消息队列，所有 Agent 读写同一个 TypedDict，简单直接
  3. 条件路由实现自适应 — should_continue 根据执行结果动态决定下一步（继续/重规划/学习），而非固定流程
  4. 记忆分层 — 短期用 buffer（快），长期用向量检索（语义匹配），情景用 SQL（结构化查询），各取所长
